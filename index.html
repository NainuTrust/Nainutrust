import random def battle_game(): player_health = 100 enemy_health = 100 inventory = ["Medkit"] print("--- Welcome to Battle Survivor (Mini Version) ---") print("Survival ka maqsad: Enemy ko harayen aur zinda rahen!\n") while player_health > 0 and enemy_health > 0: print(f"Aapki Health: {player_health} | Enemy Health: {enemy_health}") action = input("Kya karna chahenge? (attack/loot/heal): ").lower() if action == "attack": damage = random.randint(10, 25) enemy_health -= damage print(f"Aapne enemy ko {damage} damage diya!") elif action == "loot": item = random.choice(["AK-47 (Power +10)", "Shield", "Medkit"]) inventory.append(item) print(f"Aapko mila: {item}") elif action == "heal": if "Medkit" in inventory: player_health += 30 inventory.remove("Medkit") print("Aapne Medkit use ki aur 30 HP recover kar li!") else: print("Inventory mein Medkit nahi hai!") # Enemy ka turn if enemy_health > 0: enemy_damage = random.randint(5, 20) player_health -= enemy_damage print(f"Enemy ne aapko {enemy_damage} damage diya!") if player_health <= 0: print("\nGAME OVER! Aap haar gaye.") else: print("\nMUBARAK HO! Aapne enemy ko hara kar 'Booyah' kar liya!") if __name__ == "__main__": battle_game()
